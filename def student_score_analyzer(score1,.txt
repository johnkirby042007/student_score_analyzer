def student_score_analyzer(score1, score2, score3):
    average = (score1 + score2 + score3)/3

    if average >= 90:
        print("Excellent")
    elif 75 <= average <= 89:
        print("Good")
    else:
        print("Needs improvement")

    return average


score1 = float(input("Enter score 1: "))
score2 = float(input("Enter score 2: "))
score3 = float(input("Enter score 3: "))

avg = student_score_analyzer(score1, score2, score3)

print("Average:", avg)
